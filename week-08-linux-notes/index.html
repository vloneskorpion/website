:: tags:  #bash

TEST

!! # execute previous command


!n # executes the command at the history from the position n


!-n # executes command from n commands ago


!string # executes the last command starting with string

# EXAMPLE
!ls

^old^new # replaces the first occurance of old with new in the previous command and executes it

# EXAMPLE
ls -la
^la^l
# ls -l will be executed

echo $0 # name of the script
echo $num # positional parameters, representing args passed to script eg. $1
echo $# # the number of positional parameters
echo $@ # all positional parameters as separated quoted arguments
echo $* # all positional parameters as a single string
echo $? # exit status of last command executed (0 is successful)
echo $$ # the process id of current shell
echo $! # the process id of the last background command
echo $_ # the last argument from previous EXECUTED command
echo !$ # the last argument from previous command (from history), it is short for !!:$ and not eq of $_

# EXAMPLE

$ if true; then echo one; else echo two; fi
one
$ echo "$_"
one
$ if true; then echo one; else echo two; fi
$ echo !$
$ echo fi

&& # runs next command only if previous succeeds
|| # runs next command only if previous fails
; # separates multiple commands to run sequentially
